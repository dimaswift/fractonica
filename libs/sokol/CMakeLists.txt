
add_definitions(-D${sokol_backend})

# a dummy library with the system library dependencies


macro(sokol_system_deps)
    if(SOKOL_USE_WGPU_DAWN)
        fips_libs(webgpu_dawn)
    endif()
    if (FIPS_WINDOWS)
        if (SOKOL_USE_VULKAN)
            fips_libs(vulkan-1)
        endif()
    elseif (FIPS_OSX)
        fips_frameworks_osx(Foundation)
        if (FIPS_IOS)
            fips_frameworks_osx(UIKit AudioToolbox AVFoundation)
            if (SOKOL_USE_METAL)
                fips_frameworks_osx(Metal MetalKit)
            else()
                fips_frameworks_osx(OpenGLES GLKit)
            endif()
        else()
            fips_frameworks_osx(Cocoa QuartzCore AudioToolbox)
            if (SOKOL_USE_METAL)
                fips_frameworks_osx(Metal MetalKit)
            else()
                fips_frameworks_osx(OpenGL)
            endif()
        endif()
    else()
        if (FIPS_ANDROID)
            fips_libs(GLESv3 EGL log android)
            fips_libs(aaudio)
        elseif (FIPS_LINUX)
            fips_libs(X11 Xi Xcursor m dl asound)
            if (SOKOL_USE_EGL_GLES3)
                # NOTE GLESv2 is not a typo
                fips_libs(EGL GLESv2)
            elseif (SOKOL_USE_EGL_GL)
                fips_libs(EGL GL)
            elseif (SOKOL_USE_VULKAN)
                fips_libs(vulkan)
            else()
                fips_libs(GL)
            endif()
        endif()
    endif()
endmacro()

# the general sokol implementations library (compiled as C, C++ or ObjC depending on platform)
fips_begin_lib(sokol)
if (FIPS_OSX)
    fips_files(sokol.m)
else()
    fips_files(sokol.c)
endif()
sokol_system_deps()
if (FIPS_WINDOWS AND SOKOL_USE_VULKAN)
    target_include_directories(sokol PUBLIC $ENV{VULKAN_SDK}/Include)
    target_link_directories(sokol PUBLIC $ENV{VULKAN_SDK}/Lib)
endif()
if (FIPS_MSVC)
    target_compile_options(sokol PRIVATE /W4)
endif()
fips_end_lib()
